ROOT : MAIN_BODY

MAIN_BODY : DEC_FUNCTION

DEC_FUNCTION : TYPE FUNC_ID LBRACKET DEC_ARG RBRACKET LCURLY BODY RCURLY

//this contains nothing
DEC_ARG : DEC_ARG TYPE ARG
    | TYPE ARG
    |

//terminal cases
TYPE : INT
    | DOUBLE
    | STRING
    | VOID
    | CHAR
    | BOOL

//recursive with each line of code left associative
BODY : BODY STATEMENT
    | BODY

STATEMENT :  WHILE LBRACKET CONDITIONAL RBRACKET LCURLY BODY RCURLY
    | DO LCURLY BODY RCURLY LBRACKET CONDITIONAL RBRACKET SEMICOLON
